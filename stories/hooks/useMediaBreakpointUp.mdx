# useMediaBreakpointUp
Responsive utility

## Setup `App.tsx`

It is necessary to use the hook inside a DContextProvider, because the hook takes the breakpoints configuration from that context

<Source
  code={
    `
import {
  DContextProvider,
  useMediaBreakpointUpXs,
  useMediaBreakpointUpSm,
  useMediaBreakpointUpLg,
} from '@dynamic-framework/ui-react';

export function ExampleOfUse() {
  const inXs = useMediaBreakpointUpXs(true);
  const inLg = useMediaBreakpointUpLg(true);
  const inLg = useMediaBreakpointUpLg(true);

  return (
    <>
      {inXs && <p>we are in XS breakpoint</p>}
      {inSm && <p>we are in SM breakpoint</p>}
      {inLg && <p>we are in LG breakpoint</p>}
    </>
  );
}
`
  }
  language="tsx"
  dark
/>

## Setup with other settings `App.tsx`

Formatting settings can be changed by changing the values of the currency property of the DContextProvider

<Source
  code={
    `
import {
  DContextProvider,
} from '@dynamic-framework/ui-react';

import ExampleOfUse from './ExampleOfUse';

export function App() {
  return (
    <DContextProvider
      currency={{
        symbol: '$',
        precision: 2,
        separator: ',',
        decimal: '.',
      }}
    >
      <ExampleOfUse />
    </DContextProvider>
  );
}
`
  }
  language="tsx"
  dark
/>
